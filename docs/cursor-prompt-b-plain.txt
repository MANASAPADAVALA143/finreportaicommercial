Function PostToAPI(endpoint As String, jsonBody As String) As String
  - Uses MSXML2.XMLHTTP60 (works on all Excel versions, no reference needed)
  - BASE_URL = "https://your-railway-app.railway.app" (configurable constant)
  - Sets Content-Type: application/json
  - Returns response body as string
  - On error: returns JSON error string {"error": "..."} so calling code can handle gracefully

Function ReadRange(ws As Worksheet, startRow As Integer, cols As Integer) As String
  - Reads a table range into JSON array string
  - Handles: strings (quoted), numbers (unquoted), empty cells (null)
  - Returns: "[{...},{...}]" format

Function WriteResults(ws As Worksheet, jsonResponse As String, startCell As String)
  - Parses flat JSON response using InStr-based parser (no external JSON lib needed)
  - Writes key-value pairs to cells starting at startCell
  - Bold formatting for header rows, color coding for positive/negative values

Sub OpenHTMLInBrowser(htmlContent As String)
  - Saves htmlContent to %TEMP%\finreportai_output.html
  - Opens in default browser using Shell command
  - Works on Windows Excel any version

Sub ShowCommentary(commentary As String, title As String)
  - Opens a styled UserForm (frmCommentary) showing the Claude AI narrative
  - Scrollable text box, copy button, close button

VBA SUB (modPVM.bas):
Sub RunPVMAnalysis()
  Dim ws As Worksheet: Set ws = Sheets("PVM Analysis")
  
  ' 1. READ DATA from sheet
  Dim jsonData As String
  jsonData = "{""actual_revenue"": " & ws.Range("B24") & ","
  jsonData = jsonData & """budget_revenue"": " & ws.Range("B25") & ","
  jsonData = jsonData & """prior_year"": " & ws.Range("B26") & ","
  jsonData = jsonData & """products"": " & ReadRange(ws, 2, 5) & "}"
  
  ' 2. CALL API
  Application.StatusBar = "FinReportAI: Running PVM Analysis..."
  Dim response As String
  response = PostToAPI("/excel/pvm", jsonData)
  
  ' 3. WRITE RESULTS
  ' Parse and write price_effect, volume_effect, mix_effect to rows 31-34
  ws.Range("B31") = ExtractValue(response, "price_effect")
  ws.Range("B32") = ExtractValue(response, "volume_effect")
  ws.Range("B33") = ExtractValue(response, "mix_effect")
  ws.Range("B34") = ExtractValue(response, "total_variance")
  
  ' 4. WRITE COMMENTARY
  ws.Range("B37") = ExtractValue(response, "commentary")
  ws.Range("B37").WrapText = True
  
  ' 5. OPEN HTML DASHBOARD
  Dim openDash As Integer
  openDash = MsgBox("PVM analysis complete! Open interactive dashboard?", vbYesNo, "FinReportAI")
  If openDash = vbYes Then
    OpenHTMLInBrowser ExtractValue(response, "html_waterfall_dashboard")
  End If
  
  Application.StatusBar = False
  MsgBox "PVM Analysis complete!", vbInformation, "FinReportAI"
End Sub

FASTAPI ENDPOINT (routers/excel_pvm.py):
@router.post("/excel/pvm")
async def excel_pvm(data: PVMRequest):
    price_eff, vol_eff, mix_eff = calculate_pvm(data.products, data.actual_revenue, data.budget_revenue)
    commentary = await claude_commentary({
        "price_effect": price_eff, "volume_effect": vol_eff, "mix_effect": mix_eff,
        "context": "PVM waterfall analysis for CFO board presentation"
    })
    html = generate_waterfall_html(price_eff, vol_eff, mix_eff, data.actual_revenue, data.budget_revenue)
    return {"price_effect": price_eff, "volume_effect": vol_eff, "mix_effect": mix_eff,
            "total_variance": price_eff + vol_eff + mix_eff, "commentary": commentary,
            "html_waterfall_dashboard": html}

## VBA SHARED REQUIREMENTS
─────────────────────────────────────────

1. NO external VBA libraries or references required (MSXML2 is built-in)
2. Simple JSON parser using InStr/Mid — no JSON library needed
   Function ExtractValue(json As String, key As String) As String
     → finds "key": "value" or "key": number patterns
     → handles nested objects by finding key then extracting to next comma/brace
3. All subs: Show Application.StatusBar message while API call runs
4. All subs: Clear previous results before writing new ones  
5. Error handling: If API call fails, show MsgBox with helpful message
6. API timeout: Set MSXML2 timeout to 30 seconds (Monte Carlo may take longer)
7. Auth: Add X-API-Key header using value from CONFIG!B3 (optional auth key)

## FASTAPI SHARED REQUIREMENTS

1. All /excel/* endpoints are SEPARATE from /fpa/* web app endpoints
   - /excel/pvm   = lightweight, returns flat JSON (VBA-friendly)
   - /fpa/pvm     = rich response with React-formatted data (web app)
   - They can share the same calculation functions internally

2. VBA-friendly JSON responses:
   - Flat structure preferred (avoid deeply nested objects)
   - Numbers as numbers (not strings)
   - HTML content: single escaped string value

3. CORS: Already configured for web app — ensure Railway URL is also whitelisted

4. Pydantic models for all /excel/* request bodies
   - Strict validation with clear error messages (VBA shows these to user)

5. Add to main.py:
   from routers import excel_pvm, excel_three_statement, excel_monte_carlo
   from routers import excel_arr, excel_headcount, excel_board_pack, excel_sensitivity
   app.include_router(excel_pvm.router)  # etc for all 7
